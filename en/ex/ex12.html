ex12: Plot 3D contour and meshed surface with colored scaling


opt (gd:1 al:45 az:-40 gc:64) ;# gc: gradiation contour

set x = random(10000,exp,-0.5) ;# random exponential tau=0.5
set y = random(10000,gaus,5,0) ;# random gaussian sigma=5 mean=0
@ nx = 20
@ ny = 30
@ xr = "-1,10"
@ yr = "-20,20"

div 2 2

title "histogram of x"
hplot x (nb:[nx] xr:[xr])

title "histogram of y"
hplot y (nb:[ny] xr:[yr])

title "2D-histogram of x y (colored contour plot)"
hplot2 x y (nx:[nx] ny:[ny] xr:[xr] yr:[yr] ht:slope mt:cont2)

#viewport 0.05 0.95 -0.05 0.95 0.7
title "2D-histogram of x y (colored mesh plot)"
hplot2 x y (nx:[nx] ny:[ny] xr:[xr] yr:[yr] ht:slope mt:mesh2)
